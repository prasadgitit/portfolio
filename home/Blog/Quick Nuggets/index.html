OOPS Concepts:
Object Oriented Programming Concepts are important to be revised and known whether you be on which level you are. Below is a code sample that will help everyone understand basic OOPs concepts with example. 

using System;

/// 
/// 
/// OOPS 5 Building Blocks:
///     P olymorphism - one entity /name many forms
///     A bstraction - show neccessary , hide uneccessary
///     I nterface -  only blueprint of the class , 100% abstraction is achieved
///     I nheritance - gives child parent relationship b/w classes
///     E ncapsulation - encapsulate code such that 2 classes cannot see / access
///     
///Code flow as below:
/// Start
//Instantiation of class
// -Calls parameterized constructor
//	- Assigns value to x
// - Calls method1 with one parameter 
// - Calls method1 with 2 parameters
// - Calls overriden GetSubtractionResult method 
// - Write x value as 2 since we have passed 2 as constructor input while Instantiation
//Instantiate of other class
//  - Calls methods inside Interfaces
/// 
/// 


//Namespace allow creating a system to organize the code.
namespace OOPsBasics
{
    //A class or struct definition is like a blueprint that specifies what the type can do.
    struct MyStruct
    {
        public int i;
        string name;
        bool flag;
    }
    
    public class Program : myOtherClass // inheriting other class
    {
        private int x;

        #region Parameterized Constructor
        //Constructor is a special method of the class that will be automatically invoked when an instance of the class is created.
        //The main use of constructors is to initialize private fields of the class while creating an instance for the class.
        public Program(int i) // Parameterized Contructor
        {
            x = i;
        }
        #endregion

        #region Copy constructor example
        public Program(Program p) // Copy Constrcutor
        {
            x = p.x;
        }
        #endregion


        #region Main method call
        // Main method of the program to start off
        static void Main(string[] args)
        {
            Console.WriteLine("Hello World!");
            // method1("Prasad"); // this results error since non static method requires a obj ref to be created

            /*OBJECT IS AN INSTANCT OF A CLASS*/
            Program p = new Program(2); // due to parameterized constructor we need to pass a value here

            p.method1("Prasad"); // calling the non static method
            p.method1("Prasad", " Pawaskar"); // overload call to another method
            p.GetSubtractionResult(5, 4); // calling override method, this will call code line method 58 and not the actual virtual method
            //DoNothing();  // directly cannot be called proving its a non static private method by default
            Console.WriteLine(p.x); //checking constructor assignment here value would be 2
            myOtherOtherClass myOtherOtherClass = new myOtherOtherClass(); // creating instance of otherotherclass and calling its methods
            Console.WriteLine(myOtherOtherClass.GetAddtionResult(2, 3).ToString());
            Console.WriteLine(myOtherOtherClass.GetDateTimeinString().ToString());
            MyStruct myStruct = new MyStruct();
            Console.WriteLine(myStruct.i.ToString()); // only public members of struct can be accessed here
            //mySealedClass mySealedClass = new mySealedClass();
            //mySealedClass. .. Cannot access any methods since the method inside this class is protected
            //~Program(); //cannot call it this way
            //System.GC.Collect();
            //myabstractClass myabstractClass = new myabstractClass(); // this will result an error You can not create an object of Abstract Class.

        }

        #endregion

        #region Default Access modifier is private
        void DoNothing()
        {
            //blahblahblah
        }
        #endregion

        #region Polymorphism example
        //When a message can be processed in different ways it is called polymorphism.
        //Polymorphism means many forms. There are 2 types of polymorphism: Compile time polymorphism also know as Overloading Run time polymorphism also know as Overriding
        //Overloading 2 methods diff signatures same name
        private void method1(string name)
        {
            Console.WriteLine("\n " + name);
            //do something;
        }
        private void method1(string name1, string name2)
        {
            Console.WriteLine("\n " + name1 + name2);
            //do something;
        }
        //Overriding 
        //public static override int GetSubtractionResult(int x, int y) //cannot do this, static method cannot be marked as override
        //{
        //    return y - x;
        //}

        public override int GetSubtractionResult(int x, int y) //cannot do this, static method cannot be marked as override
        {
            return y - x;
        }
        #endregion

        #region destructor
        //Below is an example of desctructor
        //A Destructor is automatically invoked when an object is finally destroyed.
        // The name of the Destructor is the same as class and prefixed with a tilde(~).
        //A Destructor is used to free the dynamic allocated memory and release the resources.
        ~Program()
        {
            Console.WriteLine("\n" + "THE DESTRUCTOR IS CALLED");
            //do something
        }
        #endregion

    }

    #region interface
    //Interface example
    public interface myinterface
    {
        String GetDateTimeinString();

        //public static int GetAdditionResult(int x, int y); // this gives error since we cannot have static members in interface
        public int GetAddtionResult(int x, int y);

    }
    #endregion

    #region Virtual method
    //The virtual keyword is used to indicate that the implementation of the affected class member depends on the runtime-type of the object being invoked.
    //So when a virtual method is invoked, 
    //the runtime-type of the invoked object is checked for an overriding member.
    //The overriding member in the most-derived class is then called.
    public class myOtherClass
    {
        public virtual int GetSubtractionResult(int x, int y)
        {
            return x - y;
        }

    }
    #endregion

    #region Inheriting Interace
    //Inheritance is a process of deriving the new class from an already existing class.
    //Implementing Interface
    public class myOtherOtherClass : myinterface
    {
        public int GetAddtionResult(int x, int y)
        {
            return x + y;
        }

        public string GetDateTimeinString()
        {
            return System.DateTime.Today.ToString();
        }


    }
    #endregion

    #region Encapsulation
    //Enacpsulation example
    //Allows C# to encapsulate or wrap the code such that 2 classes cannot access it. It is achieved via access modifiers.
    sealed class mySealedClass
    {
        protected int GetMultiplyResult(int x, int y)
        {
            return x * y;
        }
    }
    #endregion

    #region Abstraction
    //Abstraction example
    //Abstraction is a process of hiding the implementation details and displaying the essential features.
    abstract class myabstractClass
    {
        public abstract void getnothing();

        public int GetDivision(int x, int y)
        {
            return x / y;
        }
    }
    #endregion

    #region Private Class
    //elements defined under the namepsace cannot cannot private class
    //private class myPrivateClass
    //{

    //}
    #endregion

    #region Static Class
    //In C#, a static class is a class that cannot be instantiated. The main purpose of using static classes in C# is to provide blueprints of its inherited classes
    static class myStaticClass //:mySealedClass  cannot inherit sealed classes in any other class
    {
        static void staticPrint()
        {
            Console.WriteLine("From static call");
        }
    }
    #endregion


    //4 access modifiers are there which makes 6 levels of accessibility.
    //1. Public: Can access entire program, class, assembly, derivied types, derived types within assembly.
    //2. Protected: within class definition and any where derived i.e.inherits
    //3. Private : Access only within class definition . default modifier if no type given
    //4. Internal: Access only to classes defined within same assembly.
    //4. a.Protected Internal: Access to current assembly and derived. 
    //3. a.Private Protected: Access limited to containing class or types derived from containing class but in same assembly.


}
